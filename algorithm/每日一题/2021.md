## 526.优美的排列

```swift
func countArrangement(_ n: Int) -> Int {
    var count = 0
    var array:[Int] = []
    var sett = Set<Int>()
    func helper(_ arr:inout [Int], _ set:inout Set<Int>){
        if arr.count >= n {
            count += 1
            return
        }
        for num in 1...n{
            if set.contains(num){
                continue
            }
            if ((arr.count+1) % num == 0) || (num % (arr.count+1) == 0) {
                arr.append(num)
                set.insert(num)
                helper(&arr,&set)
                arr.removeLast()
                set.remove(num)
            }
        }
    }
    helper(&array,&sett)
    return count
}
```

void maxheapify(vector<int>& v,vector<int>::size_type n)
{
    vector<int>::size_type max = n;
    vector<int>::size_type left = n * 2 + 1;
    vector<int>::size_type right = n* 2 + 2;
    
    if (left < v.size() && v[left] < v[max]) {
        max = left;
    }
    if (right < v.size() && v[right] < v[max]) {
        max = right;
    }
    if (max == n) {
        
        return;
    }
    int temp = v[max];
    v[max] = v[n];
    v[n] = temp;
    maxheapify(v, max);
}
vector<int> smallestK(vector<int>& arr, int k) {
   
    vector<int>::size_type halfLen = arr.size()/2;

        maxheapify(arr,i);

        re.push_back(arr[0]);

class Solution {
public:
    int largestSumAfterKNegations(vector<int>& nums, int k) {
        unordered_map<int,int> freq;
        int ans = 0;
        for(int num : nums){
            ++freq[num];
            ans += num;
        }
        for(int i=-100;i<0;++i){
            if(freq[i]){
                int ops = min(k,freq[i]);
                ans += -i * ops * 2;
                freq[i] -= ops;
                freq[-i] += ops;
                k-=ops;
                if(k==0) break;
            }
        }
        if(k > 0 && k%2 != 0){
            for(int i=0;i<=100;++i){
                if(freq[i] > 0){
                    ans = ans - 2*i;
                    break;
                }
            }
        }
        return ans;
    }

class Solution {
public:
    int findRadius(vector<int>& houses, vector<int>& heaters) {
        sort(heaters.begin(), heaters.end());
        int ans = 0;
        for(int house : houses){
            int i = upper_bound(heaters.begin(), heaters.end(),house) - heaters.begin();
            int j = i - 1;
            int rightDistance = i<heaters.size() ? heaters[i] - house : INT_MAX;
            int leftDistance = j<0 ? INT_MAX : house - heaters[j];
            int minDistance = min(rightDistance, leftDistance);
            ans = max(ans, minDistance);
        }
        return ans;
    }
};
